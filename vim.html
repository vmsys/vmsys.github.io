  68         if arg2 =~ ' ' | let arg2 = '"' . arg2 . '"' | endif
  69         let arg3 = v:fname_out
  70         if arg3 =~ ' ' | let arg3 = '"' . arg3 . '"' | endif
  71         let eq = ''
  72         if $VIMRUNTIME =~ ' '

  82         silent execute '!' . cmd . ' ' . opt . arg1 . ' ' . arg2 . ' > ' . arg3 . eq

 362 func! Compile()
 363     exe ":ccl"
 364     exe ":update"
 365     let s:Sou_Error = 0
 366     let s:LastShellReturn_C = 0
 367     let Sou = expand("%:p")
 368     let v:statusmsg = ''
 369     if expand("%:e") == "c" || expand("%:e") == "cpp" || expand("%:e") == "cxx"
 370         let Obj = expand("%:p:r").s:Obj_Extension
 371         let Obj_Name = expand("%:p:t:r").s:Obj_Extension
 372         if !filereadable(Obj) || (filereadable(Obj) && (getftime(Obj) < getftime(Sou)))
 373             redraw!
 374             if expand("%:e") == "c"
 375                 if g:iswindows
 376                     exe ":setlocal makeprg=".s:windows_CFlags
 377                 else
 378                     exe ":setlocal makeprg=".s:linux_CFlags
 379                 endif
 380                 echohl WarningMsg | echo " compiling..."
 381                 silent make
 382             elseif expand("%:e") == "cpp" || expand("%:e") == "cxx"
 383                 if g:iswindows
 384                     exe ":setlocal makeprg=".s:windows_CPPFlags
 385                 else
 386                     exe ":setlocal makeprg=".s:linux_CPPFlags
 387                 endif
 388                 echohl WarningMsg | echo " compiling..."
 389                 silent make
 390             endif
 391             redraw!
 392             if v:shell_error != 0
 393                 let s:LastShellReturn_C = v:shell_error
 394             endif
 395             if g:iswindows
 396                 if s:LastShellReturn_C != 0
 397                     exe ":bo cope"
 398                     echohl WarningMsg | echo " compilation failed"
 399                 else
 400                     if s:ShowWarning
 401                         exe ":bo cw"
 402                     endif
 403                     echohl WarningMsg | echo " compilation successful"
 404                 endif
 405             else
 406                 if empty(v:statusmsg)
 407                     echohl WarningMsg | echo " compilation successful"
 408                 else
 409                     exe ":bo cope"
 410                 endif
 411             endif
 412         else
 413             echohl WarningMsg | echo ""Obj_Name"is up to date"
 414         endif
 415     elseif expand("%:e") == "java"
 416         let class = expand("%:p:r").s:Class_Extension
 417         let class_Name = expand("%:p:t:r").s:Class_Extension
 418         if !filereadable(class) || (filereadable(class) && (getftime(class) < getftime(Sou)))
 419             redraw!
 420             exe ":setlocal makeprg=".s:JavaFlags
 421             echohl WarningMsg | echo " compiling..."
 422             silent make
 423             redraw!
 424             if v:shell_error != 0
 425                 let s:LastShellReturn_C = v:shell_error
 426             endif
 427             if g:iswindows
 428                 if s:LastShellReturn_C != 0
 429                     exe ":bo cope"

 431                 else
 432                     if s:ShowWarning
 433                         exe ":bo cw"
 434                     endif

 436                 endif

 439                     echohl WarningMsg | echo " compilation successful"
 440                 else
 441                     exe ":bo cope"
 442                 endif
 443             endif
 444         else
 445             echohl WarningMsg | echo ""class_Name"is up to date"
 446         endif
 447     else
 448         let s:Sou_Error = 1
 449         echohl WarningMsg | echo " please choose the correct source file"
 450     endif
 451     exe ":setlocal makeprg=make"

 454 func! Link()
 455     call Compile()
 456     if s:Sou_Error || s:LastShellReturn_C != 0
 457         return
 458     endif
 459     if expand("%:e") == "c" || expand("%:e") == "cpp" || expand("%:e") == "cxx"
 460         let s:LastShellReturn_L = 0
 461         let Sou = expand("%:p")
 462         let Obj = expand("%:p:r").s:Obj_Extension
 463         if g:iswindows
 464             let Exe = expand("%:p:r").s:Exe_Extension
 465             let Exe_Name = expand("%:p:t:r").s:Exe_Extension
 466         else
 467             let Exe = expand("%:p:r")
 468             let Exe_Name = expand("%:p:t:r")
 469         endif
 470         let v:statusmsg = ''
 471         if filereadable(Obj) && (getftime(Obj) >= getftime(Sou))
 472             redraw!
 473             if !executable(Exe) || (executable(Exe) && getftime(Exe) < getftime(Obj))
 474                 if expand("%:e") == "c"
 475                     setlocal makeprg=gcc\ -o\ %<\ %<.o
 476                     echohl WarningMsg | echo " linking..."
 477                     silent make
 478                 elseif expand("%:e") == "cpp" || expand("%:e") == "cxx"
 479                     setlocal makeprg=g++\ -o\ %<\ %<.o
 480                     echohl WarningMsg | echo " linking..."
 481                     silent make
 482                 endif
 483                 redraw!
 484                 if v:shell_error != 0
 485                     let s:LastShellReturn_L = v:shell_error
 486                 endif
 487                 if g:iswindows
 488                     if s:LastShellReturn_L != 0
 489                         exe ":bo cope"
 490                         echohl WarningMsg | echo " linking failed"
 491                     else
 492                         if s:ShowWarning
 493                             exe ":bo cw"
 494                         endif
 495                         echohl WarningMsg | echo " linking successful"
 496                     endif
 497                 else
 498                     if empty(v:statusmsg)
 499                         echohl WarningMsg | echo " linking successful"
 500                     else
 501                         exe ":bo cope"
 502                     endif
 503                 endif
 504             else
 505                 echohl WarningMsg | echo ""Exe_Name"is up to date"
 506             endif
 507         endif
 508         setlocal makeprg=make
 509     elseif expand("%:e") == "java"
 510         return
 511     endif
 512 endfunc

 514 func! Run()
 515     let s:ShowWarning = 0
 516     call Link()
 517     let s:ShowWarning = 1
 518     if s:Sou_Error || s:LastShellReturn_C != 0 || s:LastShellReturn_L != 0
 519         return
 520     endif
 521     let Sou = expand("%:p")
 522     if expand("%:e") == "c" || expand("%:e") == "cpp" || expand("%:e") == "cxx"
 523         let Obj = expand("%:p:r").s:Obj_Extension
 524         if g:iswindows
 525             let Exe = expand("%:p:r").s:Exe_Extension
 526         else
 527             let Exe = expand("%:p:r")
 528         endif
 529         if executable(Exe) && getftime(Exe) >= getftime(Obj) && getftime(Obj) >= getftime(Sou)
 530             redraw!
 531             echohl WarningMsg | echo " running..."
 532             if g:iswindows
 533                 exe ":!%<.exe"
 534             else
 535                 if g:isGUI
 536                     exe ":!gnome-terminal -x bash -c './%<; echo; echo 请按 Enter 键继续; read'"
 537                 else
 538                     exe ":!clear; ./%<"
 539                 endif
 540             endif
 541             redraw!
 542             echohl WarningMsg | echo " running finish"
 543         endif
 544     elseif expand("%:e") == "java"
 545         let class = expand("%:p:r").s:Class_Extension
 546         if getftime(class) >= getftime(Sou)
 547             redraw!
 548             echohl WarningMsg | echo " running..."
 549             if g:iswindows
 550                 exe ":!java %<"
 551             else
 552                 if g:isGUI
 553                     exe ":!gnome-terminal -x bash -c 'java %<; echo; echo 请按 Enter 键继续; read'"
 554                 else
 555                     exe ":!clear; java %<"
 556                 endif
 557             endif

 559             echohl WarningMsg | echo " running finish"
 560         endif
 561     endif
